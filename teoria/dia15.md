# Clase 15

### JSDocs

```javascript

/** @namespace */
var app = {};

/**
 * Resuelve la división de a/b
 * @example
 * // retorna 3
 * globalNS.metodo(5, 15);
 * @returns {Number} Retorna el resultado de la operación.
 */
app.metodo = function (a, b) {
    return b / a;
};
```
